import java.util.ArrayList;
import java.util.Arrays;

class GFG {
    public static void main(String args[])
    {
        int[] arr = new int[2];
        arr[0] = 1;
        arr[1] = 2;
        System.out.println(arr[0]);
        ArrayList<Integer> arrL = new ArrayList<Integer>(2);
        arrL.add(1);
        arrL.add(2);
        System.out.println(arrL.get(0));
    }
}
